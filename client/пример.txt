Пример кода на JQuery:

$.ajax({
	url:"ссылка к ресурсу",
	dataType:"json",
	type:"GET",
	data:{
		//данные которые надо передать
	}
	success: функцияКотораяВыполнитсяПослеУспешногоЗапроса,
	error: функцияКотораяВыполнитсяПриОшибкеЗапроса
});

function функцияКотораяВыполнитсяПослеУспешногоЗапроса(data){
	console.log(data);
	//Работа с данными, парсить уже не надо, сразу объект JS
}

function функцияКотораяВыполнитсяПриОшибкеЗапроса(header){
	console.log(header.status, header.statusText);
	alert("ААА ПАНИКА ПАНИКАААААА");
}

	способ делать запросы, но пока не настроен CORS на сервер
	этот способ будет крашиться

Пример кода на axios.js:

axios.get("ссылка к ресурсу", {
	//данные которые надо передать
})
.then( function (response) {
	//выполняется в случае удачного запроса
	console.log(response);
	//работа с данными
})
.catch( function (error) {
	//выполнится при ошибке в запросе(такие как длгий отклик, ошибка валидации, и ошибки с 
	//					кодом от 300 до 517)
	console.log("error", error)
})
